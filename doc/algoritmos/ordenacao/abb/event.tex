\begin{algorithm}
    \caption{Função \textsc{event}.} \label{abb:evento}
\begin{algorithmic}[1]
    \Function{event}{\null}
        \State $e_1 \leftarrow  $ \Call{minPQ}{$Q$}
        \While{$e_1.cert$ = \now}
            \State $e_2 \leftarrow e_1.prev$
            \State \Call{swap}{$e_1, e_2$}
            \State \Call{update}{$e_1$}
            \State \Call{update}{$e_2$}
            \State \Call{update}{$e_2.next$}
            \State $e_1 \leftarrow $ \Call{minPQ}{$Q$}
        \EndWhile
    \EndFunction
\end{algorithmic}
\end{algorithm}
