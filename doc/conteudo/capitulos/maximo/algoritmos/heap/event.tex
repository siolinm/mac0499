    \caption[Algoritmo \textsc{event} do heap cinético]{Função \textsc{event}.} \label{alg:heap:evento}
    \begin{algorithmic}[1]
        \Function{event}{\null}
            \State $i \leftarrow  $ \Call{minPQ}{$Q$}

                \heap$[\floor{\frac{i}{2}}]$

                \floor{\frac{i}{2}} + ((i + 1)~mod~2), 2i, 2i+1\rk$}
                    \State \Call{update}{$j$}
                \EndFor

                \State $i \leftarrow  $ \Call{minPQ}{$Q$}
            \EndWhile
        \EndFunction
    \end{algorithmic}
