\begin{algorithm}
    \caption[Algoritmo \textsc{query\_kth} da lista ordenada]{Função \textsc{query\_kth}.} \label{alg:lista:query}

        \Function{query\_kth}{$i$}
            \If{$1 \leq i \leq n$}

            \EndIf
            \State \Return $-1$
        \EndFunction
