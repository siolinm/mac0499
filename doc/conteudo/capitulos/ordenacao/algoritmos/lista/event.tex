\begin{algorithm}[H]
    \caption[Algoritmo \textsc{event} da lista ordenada]{Função \textsc{event}.} \label{alg:lista-ordenada:evento}
    \begin{algorithmic}[1]
        \Function{event}{$i$}
            \State \sorted$[i] \leftrightarrow$
            \sorted$[i + 1]$
            \State $indS[\sorted[i]] \leftrightarrow
            indS[\sorted[i+1]]$
            \State \Call{update}{$i$}
            \State \Call{update}{$i - 1$}
            \State \Call{update}{$i + 1$}
        \EndFunction
    \end{algorithmic}
\end{algorithm}
