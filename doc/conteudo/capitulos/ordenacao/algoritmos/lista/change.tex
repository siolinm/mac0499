\begin{algorithm}[H]
    \caption[Algoritmo \textsc{change} da lista ordenada]{Função \textsc{change}.} \label{alg:lista-ordenada:change}
    \begin{algorithmic}[1]
        \Function{change}{$j, v$}
            \State $x_0$[$j$] $\leftarrow  x_0$[$j$]
            $+~($\speed[$j$]$~-~v)~\cdot~$\now;
            \State \speed[$j$] $\leftarrow  v$
            \State $i \leftarrow$ \inds[$j$]
            \State \Call{update}{$i$}
            \State \Call{update}{$i - 1$}
        \EndFunction
    \end{algorithmic}
\end{algorithm}
