\begin{algorithm}
    \caption[Algoritmo \textsc{change} da árvore binária de busca]{Função \textsc{change}.} \label{alg:abb:change}
    \begin{algorithmic}[1]
        \Function{change}{$j, v$}
            \State $e \leftarrow$ \Call{getObject}{$j$}
            \State $e.x_0$ $\leftarrow  e.x_0+~($\speed[$j$]$~-~v)~
            \cdot~$\now;
            \State $e.speed\leftarrow v$
            % \State $i \leftarrow$ \inds[$j$]
            \State \Call{update}{$e$}
            \State \Call{update}{$e.next$}
        \EndFunction
    \end{algorithmic}
\end{algorithm}
