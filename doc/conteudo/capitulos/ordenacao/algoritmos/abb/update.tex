\begin{algorithm}
    \caption{Função \textsc{update}.} \label{abb:update}
\begin{algorithmic}[1]
    \Function{update}{$e$}
        \If{$e \neq$ NULL}
            \State $e'\leftarrow e.next$
            \State $t \leftarrow $ \Call{expire}{$e,e'$}
            \State \Call{updatePQ}{$Q,e,t$}
        \EndIf
    \EndFunction
\end{algorithmic}
\end{algorithm}
