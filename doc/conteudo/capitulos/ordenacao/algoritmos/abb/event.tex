\begin{algorithm}
    \caption[Algoritmo \textsc{event} da árvore binária de busca]{Função \textsc{event}.} \label{alg:abb:evento}
    \begin{algorithmic}[1]
        \Function{event}{$e$}
            \State $e_p \leftarrow e.prev$
            \State \Call{swap}{$e, e_p$}
            \State \Call{update}{$e$}
            \State \Call{update}{$e_p$}
            \State \Call{update}{$e_p.next$}
            \State $e \leftarrow $ \Call{minPQ}{$Q$}
        \EndFunction
    \end{algorithmic}
\end{algorithm}
