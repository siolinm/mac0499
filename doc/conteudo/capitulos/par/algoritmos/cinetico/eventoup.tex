\begin{algorithm}[H]
    \caption[Algoritmo \textsc{upEvent} do par mais próximo cinético]{Função \textsc{upEvent}.}
    \label{alg:par-cinetico:eventoup}
    \begin{algorithmic}[1]
        \Function{upEvent}{$p, q, dir$}
            \If{$q = \Call{owner}{p.hitsLow(dir)}$}
                \Comment{$p$ está em $\Hits_{low}(q)$}
                \State \Call{upEventLeft}{$p, q, dir$}
            \Else
                \If{$p = \Call{owner}{q.cands(dir)}$}
                    \Comment{$q \in \Cands(p)$}
                    \State \Call{upEventRight}{$p, q, dir$}
                \EndIf
            \EndIf
        \EndFunction
    \end{algorithmic}
\end{algorithm}
